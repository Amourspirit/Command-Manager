The followng is Code segments from the Command TextEditor Example Application that handles cut/copy and paste for the text editor.


In the sample code note that each Command is handeling a toolbar button, Main Menu Item and a Popup Menu Item all without the use of the any of the Menu or button event handlers. Other controls could easisly be added by adding simply adding the control to the AddRange method.


Sample Code:

  1. public partial class TextEditor : Form
  2. {
  3.     // constructor
  4.     public TextEditor()
  5.     {
  6.         InitializeComponent();
  7.         InitializeCommandManager();
  8.     }
  10.     private void InitializeCommandManager()
  11.     {
  12.         cmdMgr.Commands.Add(new Command("CmdEditCopy", new Command.ExecuteHandler(CmdEditCopy)
  13.             , new Command.UpdateHandler(CmdEditCopyHandler)));
  14.         cmdMgr.Commands["CmdEditCopy"].CommandInstances.AddRange(mnuEditCopy, cMnuEditCopy, tb_Copy);
  15.         cmdMgr.Commands.Add(new Command("CmdMainCut", new Command.ExecuteHandler(CmdEditCut)
  16.             , new Command.UpdateHandler(CmdEditCutHandler)));
  17.         cmdMgr.Commands["CmdMainCut"].CommandInstances.AddRange( mnuEditCut, cMnuEditCut, tb_Cut);
  18.         cmdMgr.Commands.Add(new Command("CmdMainPaste", new Command.ExecuteHandler(CmdEditPaste)
  19.             , new Command.UpdateHandler(CmdEditPasteHandler)));
  20.         cmdMgr.Commands["CmdMainPaste"].CommandInstances.AddRange(mnuEditPaste, cMnuEditPaste , tb_Paste);
  21.     }
  23.     #region Command Handlers
  24.     #region Cut / Copy / Paste / Handlers
  25.     private void CmdEditCutHandler(Command Cmd)
  26.     {
  27.         Cmd.Enabled = (this.txtDocument.Enabled
  28.         && this.txtDocument.SelectionLength > 0);
  29.     }
  31.     private void CmdEditCopyHandler(Command Cmd)
  32.     {
  33.         Cmd.Enabled = (this.txtDocument.SelectionLength > 0);
  34.     }
  36.     private void CmdEditPasteHandler(Command Cmd)
  37.     {
  38.         Cmd.Enabled = Clipboard.GetDataObject().GetDataPresent(typeof(string));
  39.     }
  40.     #endregion
  41.     #endregion
  43.     #region Command Methods
  44.     #region Cut / Copy / Paste / Mehods
  45.     private void CmdEditCopy(Command Cmd)
  46.     {
  47.         Clipboard.SetText(this.txtDocument.SelectedText);
  48.     }
  50.     private void CmdEditCut(Command Cmd)
  51.     {
  52.         Clipboard.SetText(txtDocument.SelectedText);
  53.         txtDocument.SelectedText = string.Empty;
  54.     }
  56.     private void CmdEditPaste(Command Cmd)
  57.     {
  58.         txtDocument.Paste();
  59.     }
  60.     #endregion
  61.     #endregion
  62.     // ... other code
  63. }
